cov = [[0.1, 0], [0, 0.1]]
x = []
y = np.random.randint(2, size=(total_num))
y0 = []
y1 = []
for i in range(total_num):
    mean = [y[i], y[i]]
    point = (np.random.multivariate_normal(mean, cov, 1))[0]
    x.append(point)
    if y[i] == 0:
        y0.append(point)
    else:
        y1.append(point)

x = np.array(x)
x0 = x[:, 0]
